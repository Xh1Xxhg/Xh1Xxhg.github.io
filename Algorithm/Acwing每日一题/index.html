# 二分 & 差分

AcWing 503. 借教室

数据范围 10 的 6 次方 => 时间复杂度 O(n)或小常数O(nlogn)

  • 总体思路
    • 二分（查找）：设置 mid 为最后一天没出现负值，mid+1 为第一天出现负值
    • 差分（判断）：计算每天的剩余教室量，判断是否 <u> 满足最后一天没出现负值 </u > 的条件
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long LL;

const int N = 1000010;

int n,m;
int w[N];
int d[N],l[N],r[N];
// 差分数组
LL b[N];

bool check(int mid) {
    memset(b, 0, sizeof b);
    // 差分数组状态更新
    for (int i = 1; i <= mid; i ++ ) {
        b[l[i]] += d[i];
        b[r[i] + 1] -= d[i];
    }
    for (int i = 1; i <= n; i ++ ) {
        b[i] += b[i - 1];
        if(b[i] > w[i]) return false;
    }
    return true;
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i ++ )
        scanf("%d", &w[i]);
    for (int i = 1; i <= m; i ++ )
        scanf("%d%d%d", &d[i], &l[i], &r[i]);
        
    // l=0表示一个订单都不满足
    int l = 0, r = m;
    while (l < r) {
        int mid = (l + r + 1) >> 1;
        if (check(mid)) l = mid;
        // mid不满足条件需要舍弃
        else r = mid - 1;
    }
    
    if (r == m) puts("0");//没动
    // l=r=mid,mid满足条件，选择不满足条件的mid+1
    else printf("-1\n%d\n", r + 1);
    
    return 0;
}

# 前缀和

AcWing 562. 壁画

数据范围 10 的 6 次方 => 时间复杂度 O(n)或小常数O(nlogn)

  • 画的墙是连续的一段
  • 毁坏的墙只出现在两端
  • 画一个坏一个这样的顺序模式，导致最终长度一定是 ⌈n/2⌉上取整
  • 可以取中间部分的墙体，以中点划分，被画的部分长分别为 a,b ，定制规则：第一次画中点的墙体，然后哪边毁了画哪边，n 长的墙体分为 a,a-1,1,b,b
  • 前缀和计算
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 5000010;

int n;
int s[N];
char str[N];

int main()
{
    // T组测试数据
    int T;
    scanf("%d", &T);
    
    // 按组别数循环
    for (int cases = 1; cases <= T; cases ++ ) {
        // 墙体长度
        scanf("%d", &n);
        // 评分标准序列
        scanf("%s", str + 1);
        for (int i = 1; i <= n; i++ )
            // 递进计算每次画墙后的总分
            s[i] = s[i - 1] + (str[i] - '0');
            
        // 从中点开始画
        int res = 0, m = (n + 1) / 2;
        for (int i = m; i <= n; i ++ )
            // 计算前缀和：对半移动
            res = max(res, s[i] - s[i - m]);
        printf("Case #%d: %d\n", cases, res);
    }
    return 0;
}

# 差分

AcWing 4262. 空调

数据范围 10 的 5 次方 => 时间复杂度 O(nlogn)

  • 题意变为差值数组，T=>P 改为 0=>T-P
  • 利用差分，一段范围内的更新变为两点 b[l]++,b[r+1]-- 更新
  • 下界：差分数组和为 0，正数负数可以相加抵消 => 最小的指令次数≥差分数组整数之和，这个下界值即为答案最小数量

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100010;
int n;
int b[N];

int main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &b[i]);
    // 算出转换的差值数组
    for (int i = 1; i <= n; i ++ ) {
        int m;
        scanf("%d", &m);
        b[i] -= m;
    }
    // 差分数组
    // 倒序改变值，因为d[]和b[]共用一个数组顺序解决会提前改变b[l]值，导致b[r + 1]值变化
    // b[l] = d[l] - d[l - 1]
    // b[r + 1] = d[r + 1] - d[r]
    for (int i = n + 1; i; i -- ) b[i] -= b[i - 1];
    
    // 算出差分数组的正数之和，即为解的下界
    int res = 0;
    for (int i = 1; i <= n + 1; i ++ )
        if(b[i] >= 0)
            res += b[i];
    printf("%d", res);
    return 0;
}

# 双指针

AcWing 3745. 牛的学术圈 I

数据范围 10 的 5 次方 => 时间复杂度 O(nlogn)

  • 需要按照引用次数计算 h 指数，先排序，再考虑条件
    • 前 h 篇论文的引用次数都大于等于 h-1（写完综述 + 1 变 h 次）
    • 其中最多有 L 个数等于 h-1（最多引用）
  • 核心思路：排序后数组为单调递减（ i篇引用≥h，引用次数Ci越大，下标i越靠前 ），找到大于等于 h 的最后一个数，获取下标 j ，判断是否下标 j<=L ，同时是否存在 h-j篇引用次数为h-1
  • 一个下标关于另一个下标单调 => 双指针法
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100010;
int n,L;
int w[N];

int main()
{
    scanf("%d%d", &n, &L);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &w[i]);
    
    sort(w + 1, w + n + 1, greater<int>());
    
    
    int res = 0;
    for (int i = 1, j = n; i <= n; i ++ ) {
        // j没遍历完，同时保证w[j]>=h
        while (j && w[j] < i) j--;
        // i遍历完所有h-1停止，同时保证h-1个数<=L
        if (w[i] >= i - 1 && i - j <= L)
            res = i;
    }
    printf("%d", res);
    return 0;
}

# 归并排序

AcWing 505. 火柴排队

  • 需要解决的问题
    • 什么情况下距离最小：A、B 都同时以升序 / 降序排列
    • 如何通过顺序交换实现这个距离： 离散化=>映射=>归并排序(代码模板)=>逆序对
  • 火柴高度（数值过大，用作数组下标时，可能会超内存）：离散化（只考虑相对大小关系），将 A 和 B 都变为 1,...,n 的排列分布， 离散化是一种将原始数据映射到连续的整数集合的过程，通常用于处理离散值范围较大的情况
  • 离散化方法
    • 排序 + 二分
    • lambda 表达式法
  • 简化问题：A 有序，只能交换 B（因为每次只能交换相邻数，所以最小交换次数是逆序对，求逆序对 => 归并排序 、 树状数组 ），若只能交换 A，也类似
  • 回归问题，A，B 都无序，但是因为 离散化后分布一致 ，利用映射的思想，将 A 中的数映射为升序，只需要将 B 中与之对应的数映射变换即可，生成的新序列与原序列在本质上别无二致

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100010, MOD = 99999997;
int n;
// 
int a[N], b[N], c[N], p[N];

// a,b离散化
void work(int a[]) {
    // 存入顺序
    for (int i = 1; i <= n; i ++ ) p[i] = i;
    sort(p + 1, p + n + 1, [&](int x, int y){
        return a[x] < a[y];
    });
    // 离散化实现
    for (int i = 1; i <= n; i ++ ) a[p[i]] = i;
}

// 归并排序计算逆序对
int merge_sort(int l, int r) {
    if (l >= r) return 0;
    int mid = (l + r) >> 1;
    // 分别递归调用左右部分，逆序对相加取模
    int res = (merge_sort(1, mid) + merge_sort(mid + 1, r) )% MOD;
    
    int i = 1, j = mid + 1, k = 0;
    // 较小的元素放入辅助数组p[]
    while (i <= mid && j <= r) {
        if (b[i] < b[j]) p[k ++ ] = b[i ++ ];
        // res增加mid - i + 1，表示左半部分剩余的元素都大于b[j]
        else p[k ++ ] = b[j ++ ], res = (res + mid - i + 1) % MOD;
    }
    // 剩余的元素复制回原数组b[]
    while (i <= mid) p[k ++ ] = b[i ++ ];
    while (j <= r) p[k ++ ] = b[j ++ ];
    
    for (int i = 1, j = 0; j < k; i ++ , j ++ ) b[i] = p[j]; 
    return res;
}

int main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &a[i]);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &b[i]);
    
    work(a), work(b);
    // c存入a的顺序映射关系
    for (int i = 1; i <= n; i ++ ) c[a[i]] = i;
    // b按照c存入的映射关系完成影身
    for (int i = 1; i <= n; i ++ ) b[i] = c[b[i]];
    
    // 利用归并排序求得逆序对
    printf("%d\n", merge_sort(1, n));
    
    
    return 0;
}

# 多路归并

AcWing 1262. 鱼塘钓鱼

  • 贪心：路程时间 + 钓鱼时间（一定是直线走法，不会迂回）
  • 多路归并：多个单调递减的鱼塘序列合并为一个有序序列， 从当前每个鱼塘的数量中选择最大的
#include <algorithm>
#include <iostream>
#include <cstring>

using namespace std;

const int N = 110;
// 鱼的数量，每分钟钓鱼数，两鱼塘间隔时间，每个鱼塘的钓鱼停留时间
int num[N], dect[N], t[N], spend[N];

// 鱼塘剩余的鱼数量
int get(int k) {
    return max(0, num[k] - dect[k] * spend[k]);
}

// work(上一个满足方案的鱼塘编号, 剩余钓鱼次数1次/min)
int work(int n, int ddl) {
    int res = 0;
    // 初始化每个鱼塘钓鱼时间为0
    memset(spend, 0, sizeof spend);
    
    // 剩余钓鱼次数
    for (int i = 0; i < ddl; i ++ ) {
        // 从第一个开始和后面每个鱼塘比较，找出最大
        int t = 1;
        for (int j = 2; j <= n; j ++ )
            if (get(t) < get(j))
                t = j;
        res += get(t);
        spend[t] ++ ;
    }
    return res;
}

int main()
{
    int n, ddl;
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &num[i]);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &dect[i]);
    for (int i = 2; i <= n; i ++ ) {
        scanf("%d", &t[i]);
        t[i] += t[i - 1];
    }
    scanf("%d", &ddl);
    
    int res = 0;
    // 从第一个鱼塘开始找到最大的钓鱼数
    for (int i = 1; i <= n; i ++ )
        res = max(res, work(i, ddl - t[i]));
    
    printf("%d\n", res);
    
    return 0;
}

# 贡献法

AcWing 4261. 孤独的照片

  • 先枚举区间 起点+终点 ，再枚举牛， O(n^2) ；先枚举牛 ( 每头牛对孤独区间的贡献 )，再枚举区间，找到牛对应的孤独区间， O(n)

  • 枚举牛：获得每头牛位置之前的子区间中以 G 为分隔符的最大长度

  • 连续牛的个数不少于

    • 孤独的牛 G 左右两边至少有一个另一种牛 H，LR 个可能
    • 孤独的牛 G 左无 H，右边至少两个 H， R-1 个可能
    • 孤独的牛 G 右无 H，左边至少两个 H， L-1 个可能
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 500010;
typedef long long LL;

int n;
char str[N];
int l[N], r[N];

int main() {
    scanf("%d%s", &n, str);
    // l[],r[]从左到右和从右到左每个位置的子串中以 'G' 为分隔符的最大长度
    // sh, sg表示当前位置之前H或G连续的个数
    for (int i = 0, sh = 0, sg = 0; i < n; i ++ ) {
        if (str[i] == 'G') l[i] = sh, sg ++ , sh = 0;
        else l[i] = sg, sh ++ , sg = 0;
    }
    
    for (int i = n -1, sg = 0, sh = 0; i >= 0; i -- ) {
        if(str[i] == 'G') r[i] = sh, sg ++ , sh = 0;
        else r[i] = sg, sh ++ , sg = 0;
    }
    
        
    LL res = 0;
    for (int i = 0; i < n; i ++ )
        // 给(LL)加括号，提前进行类型转换
        res += (LL)l[i] * r[i] + max(r[i] - 1, 0) + max(l[i] - 1, 0);
    
    printf("%lld\n", res);
    
    return 0;
}

更易理解做法

#include <bits/stdc++.h>
using namespace std;

const int N = 500010;
typedef long long LL;

int n;
char s[N];

int main()
{
    cin >> n >> s; // 读入字符串长度和字符串内容

    LL res = 0; // 用来存储结果的变量，初始化为0

    for (int i = 0; i < n; i ++) // 遍历字符串中的每个位置
    {
        char x = s[i]; // 当前位置的字符
        int l = i - 1, r = i + 1; // 初始化左右两个指针

        // 向左寻找以当前字符为分隔符的最长子串
        while (l >= 0 && x != s[l]) l --;
        // 向右寻找以当前字符为分隔符的最长子串
        while (r < n && x != s[r]) r ++;

        int L = i - l - 1, R = r - i - 1; // 计算左右两侧子串的长度
        res += (LL)L * R; // 更新结果，累加以当前位置为分隔符的子串对数
        if (L > 0) res += L - 1; // 如果左侧子串长度大于0，加上左侧子串内部的分隔符对数
        if (R > 0) res += R - 1; // 如果右侧子串长度大于0，加上右侧子串内部的分隔符对数
    }

    cout << res << endl; // 输出结果

    return 0; 
}

# 日期问题

AcWing 3498. 日期差值

  • 数据范围：按天枚举，100 组 10000 年，估摸 3.65 亿次，有超时的风险
  • 思路变换为
    • 两个时间的差值 = abs (第一个时间到公元第一天天数 - 第二个时间到公元第一天的天数)

这里需要注意，因为有 100 组数据，需要用 while 循环读数据

scanf返回值：
如果只有a被成功读入，返回值为1；
如果a和b都未被成功读入，返回值为0；
如果遇到错误或遇到end of file（文件结束），返回值为EOF（即-1）。
~的意义：EOF（-1）补码表示为全1，按位取反后为全0，即false
~scanf表示while循环在读完所有组别数据后就终止循环

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int month[] = {
    0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31
};

int is_leap(int year) {
    if (year % 4 == 0 && year % 100 || year % 400 == 0)
        return 1;
    return 0;
}

// 获取每个月天数
int get_days(int y, int m) {
    if (m == 2) return 28 + is_leap(y);
    return month[m];
}

int calc(int y, int m, int d) {
    int res = 0;
    for (int i = 1; i < y; i ++ )
        res += 365 + is_leap(i);
    for (int i = 1; i < m; i ++ )
        res += get_days(y, i);
    return res + d;
}


int main()
{
    int y1, y2, m1, m2, d1, d2;
    while (~scanf("%04d%02d%02d\n%04d%02d%02d", &y1, &m1, &d1, &y2, &m2, &d2))
        printf("%d\n", abs(calc(y1, m1, d1) - calc(y2, m2, d2)) + 1);
    return 0;
}

# 区间合并

AcWing 1343. 挤牛奶

数据范围 10 的 5 次方 => 时间复杂度 O(nlogn)

#include <bits/stdc++.h>
#define x first
#define y second

using namespace std;

typedef pair<int,int> PII;

const int N = 5010;
int n;
PII segs[N];

int main() {
    // 输入区间个数
    scanf("%d", &n);

    // 输入各个区间的起始和结束点，并存储在数组segs中
    for (int i = 0; i < n; i ++ ) 
        scanf("%d%d", &segs[i].x, &segs[i].y);

    // 按区间的起始点进行升序排序
    sort(segs, segs + n);

    // 初始化最长连续区间和最长间隔区间的结果为0
    int res1 = 0, res2 = 0;

    // 初始化当前连续区间的左右边界为第一个区间的起始和结束点
    int l = segs[0].x, r = segs[0].y;

    // 遍历区间数组
    for (int i = 1; i < n; i ++ )
        // 如果当前区间的起始点小于等于当前连续区间的右边界，则合并区间
        if (segs[i].x <= r) 
            r = max(r, segs[i].y);
        else {
            // 更新最长连续区间和最长间隔区间的结果
            res1 = max(res1, r - l);
            res2 = max(res2, segs[i].x - r);
            // 更新到下一组区间，更新左右边界
            l = segs[i].x, r = segs[i].y;
        }

    // 更新最后一组区间的结果
    res1 = max(res1, r - l);

    // 输出最长连续区间和最长间隔区间的结果
    printf("%d %d\n", res1, res2);

    return 0;
}

#include <iostream>
#include <cstring>
#include <algorithm>

#define x first
#define y second

using namespace std;

const int N = 180;

typedef pair<int, int> PII;

int n;
PII q[N * N];

// 判断互质
int gcd(int a, int b) {
    // 辗转相除法
    return b ? gcd(b, a % b) : a;
}

bool cmp(PII a, PII b) {
    return a.y * b.x < a.x * b.y;
}

int main()
{
    scanf("%d",&n);
    
    int cnt = 0;
    // 分母
    for (int i = 0; i <= n; i ++ )
        // 分子
        for (int j = 0; j <= i; j ++ )
            if (gcd(i, j) == 1)
                q[cnt ++] = {i, j};
                
    sort(q, q + cnt, cmp);
    
    // {i, j} => {分母，分子} <= {x, y}
    for (int i = 0; i < cnt; i ++ )
        printf("%d/%d\n", q[i].y, q[i].x);
    
    return 0;
}

    printf("%d/%d\n", a + c, b + d);
    // 右半边递归，分子分母相加
    dfs(a + c, b + d, c, d);
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    
    puts("0/1");
    dfs(0, 1, 1, 1);
    puts("1/1");
    
    return 0;
}

# DFS

  • 转换为计算两个由 X 组成的矩阵中所有 X 之间的最小曼哈顿距离（两个点横坐标差的绝对值和纵坐标差的绝对值的和）
  • 矩阵 N 行 M 列，矩阵大小 50*50=2500 ，遍历面积的最大格式为 c(2500-c) ，根据均值不等式，得知在中点处最大，数据量大小为 100 万左右
  • 核心思路：
    • 枚举，找到一个 X 时，用 DFS/BFS/并查集 搜索出整个连通块（DFS 代码短；BFS 第一次求得为最短路径，栈空间较小）
  • vector<PII>& q 通过引用允许函数直接操作调用者提供的变量，而不是对变量进行复制，这样可以减少内存消耗和提高效率
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;
const int N = 55;

int n, m;
char g[N][N];// 牛皮矩阵
vector<PII> points[2];// 保存点坐标{[x1, x2, ...], [y1, y2, ...]}
// 4方向偏移量
int dx[4] = {-1, 0, 1, 0}, dy[4] = {0, 1, 0, -1};

void dfs(int a, int b, vector<PII>& q) {
    // 入栈
    q.push_back({a, b});
    g[a][b] = '.';
    
    for (int i = 0; i < 4; i ++ ) {
        // 更新纵坐标和横坐标
        int x = a + dx[i], y = b + dy[i];
        // 不越界继续搜索，复原出斑点的位置
        if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == 'X')
            dfs(x, y, q);
    }
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; i ++ ) scanf("%s", g[i]);
    
    // 连通块的个数
    int k = 0;
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            if (g[i][j] == 'X')
                dfs(i, j, points[k ++ ]);
    // 最远的曼哈顿距离需要>100，这里设置为110
    int res = 110;
    // 遍历获取斑点的坐标值
    for (auto& a: points[0])
        for (auto& b: points[1])
            res = min(res, abs(a.x - b.x) + abs(a.y - b.y)); // 更新最小曼哈顿距离
            
    printf("%d\n", res - 1);
    return 0;
}

# 剪枝

AcWing 167. 木棒

  • 暴搜：从小到大枚举答案的长度 len ，依次拼每根长度为 len 的木棍
  • 剪枝优化
    • 优化搜索顺序：从大到小去枚举，减少了后续木棍的选择，进而减少了搜索树的节点数目
    • 排除等效冗余：组合型枚举，每一根长木棍的内部的小木棍编号递增
    • 可行性剪枝
      • 失败后，跳过与当前木棍长度相等的其他木棍
      • 某个新长木棍中，尝试接入的第一根木棍的递归分支返回失败的话，则回溯更换
      • 若当前小木棍接入的是最后一根符合条件的长木棍，剩余木棍无解，则回溯更换
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 60;

int n;
int stick[N]; // 每根木棍的长度
bool st[N]; // 木棍是否被使用
int sum, len; // sum当前木棍的总长度；len枚举的长度

// u当前枚举到的长木棍;cur当前长木棍的长度;start当前长木棍在拼的小木棍的下标
bool dfs(int u, int cur, int start) {
    // 所有的小木棍都拼成了长木棍
    if (u * len == sum) return true;
    // 长木棍已经拼接完成，搜索下一根
    if (cur == len) return dfs(u + 1, 0, 0);
    
    for (int i = start; i < n; i ++ ) {
        // 当前小木棍被用过了
        if (st[i]) continue;
        // 判断当前木棍能否使用
        if (cur + stick[i] <= len) {
            st[i] = true;
            if (dfs(u, cur + stick[i], i + 1)) return true;
            st[i] = false;
        }
        
        // 当前是一根新长木棍，剪枝2和3
        if (!cur || cur + stick[i] == len) return false;
        
        // 剪枝1，从i+1开始，长度与i相等的跳过
        int j = i +  1;
        while (j < n && stick[j] == stick[i]) j++;
        i = j - 1;
    }
    return false;
}

int main()
{
    while (cin >> n, n) {
        // 清空
        sum = len = 0;
        // 依次读入每根木棍
        for (int i = 0; i < n; i ++ ) {
            cin >> stick[i];
            sum += stick[i];
            len = max(len, stick[i]); // 最小的长度要大于等于最短木棍的长度
        }
        // 从大到小排序
        sort(stick, stick + n, greater<int>());
    
        // 清空
        memset(st, 0, sizeof st);
        while (true) {
            // 整除且能够搜索到方案
            if (sum % len == 0 && dfs(0, 0, 0)) {
                // 输出枚举长度
                cout << len << endl;
                break;
            }
            len ++ ;
        }
    }
    return 0;
}

# BFS

AcWing 1355. 母亲的牛奶

  • 数据范围：1≤A,B,C≤20，最大范围 21^3 个点
  • 状态转移：每一个点为三元组，记录每一桶奶的存量，每次转移都只有 6 种选择

解法一

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 21, M = N * N * N;

int A, B, C;
struct Node
{
    int a, b, c;
}q[M];
bool st[N][N][N];

void bfs()
{
    // 头尾
    int hh = 0, tt = 0;
    q[0] = {0, 0, C}; //初始桶的状态
    st[0][0][C] = true;
    int W[3] = {A, B, C}; // 三个桶容量
    
    // 广搜BFS
    while (hh <= tt)
    {
        auto t = q[hh ++ ]; // Node类型
        for (int i = 0; i < 3; i ++ )
            for (int j = 0; j < 3; j ++ )
                if (i != j) // 从桶i往通j中倒
                {
                    int w[3] = {t.a, t.b, t.c}; // 记录此时奶量状态
                    int r = min(w[i], W[j] - w[j]); // 不超过i现有的奶量，不超过j剩余存量
                    w[i] -= r, w[j] += r; // 奶量状态改变
                    int a = w[0], b = w[1], c = w[2];
                    if (!st[a][b][c]) // 当前状态没被拓展
                    {
                        st[a][b][c] = true;
                        q[ ++ tt] = {a, b, c};
                    }
                }
    }
    
}

int main()
{
    scanf("%d%d%d", &A, &B, &C);
    bfs();
    
    //  A桶为空
    for (int c = 0; c <= C; c ++ )
        for (int b = 0; b <= B; b ++ )
            if (st[0][b][c])
            {
                printf("%d ", c);
                break;
            }
    
    return 0;
}

解法二：对每次桶内奶量转移的方式进行递归

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 21, M = N * N * N;

int A, B, C;
struct Node
{
    int a, b, c;
}q[M];
bool st[N][N][N];
int hh, tt;

void Insert(int a, int b, int c) 
{
    if(!st[a][b][c])
    {
        q[ ++ tt] = {a, b, c};
        st[a][b][c] = true;
    }
}

void bfs()
{
    q[0] = {0, 0, C};
    st[0][0][C] = true;
    while (hh <= tt) 
    {
        auto t = q[hh ++ ];
        int a = t.a, b = t.b, c = t.c;
        Insert(a - min(a, B - b), min(a + b, B), c); // a->b
        Insert(a - min(a, C - c), b, min(a + c, C)); // a->c
        Insert(min(a + b, A), b - min(b, A - a), c); // b->a
        Insert(a, b - min(b, C - c), min(b + c, C)); // b->c
        Insert(min(a + c, A), b, c - min(c, A - a)); // c->a
        Insert(a, min(b + c, B), c - min(c, B - b)); // c->b
    }
}

int main()
{
    scanf("%d%d%d", &A, &B, &C);
    bfs();
    
    //  A桶为空
    for (int c = 0; c <= C; c ++ )
        for (int b = 0; b <= B; b ++ )
            if (st[0][b][c])
            {
                printf("%d ", c);
                break;
            }
    
    return 0;
}

# Flood Fill

AcWing 687. 扫雷

  • 数据范围：100 * 300 * 300，不到 10000000，线性时间复杂度

  • 思路

    • 点一遍每个 0 的连通块，DFS 搜索出来
    • 1-8 + 周围无 0 必点
    • 1-8 + 周围有 0 不点（先点 0 再点 1，就不需要点 1，因为 1 是周围自动展开的）

  • 答案：0 的连通块的数量 + 周围无 0 的 1-8 的数量

#include <algorithm>
#include <cstring>
#include <iostream>

using namespace std;

const int N = 310;

int n;
char str[N][N]; // 存储矩阵
int g[N][N];    // 存储每个格子周围的数量

// Flood Fill算法
void dfs(int a, int b) {
    int t = g[a][b]; // 标记为不是0或者1-8的数
    g[a][b] = -1; // 经过遍历
    if (t) return; // 扩展到非0数字，即边界，修改状态为-1，停止扩展

    // 判断周围八个位置（避免越界）
    for (int x = a - 1; x <= a + 1; x ++ )
        for (int y = b - 1; y <= b + 1; y ++ )
            if (x >= 0 && x < n && y >= 0 && y < n && g[x][y] != -1)
                dfs(x, y);
}

int main()
{
    int T; // 测试数据组数
    scanf("%d", &T);

    for (int cases = 1; cases <= T; cases ++ )
    {
        scanf("%d", &n); // 输入矩阵大小(游戏网格)
        for (int i = 0; i < n; i++) scanf("%s", str[i]);

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                // 有雷
                if (str[i][j] == '*') g[i][j] = -1;
                // 统计周围八个方向的雷数
                else
                {
                    g[i][j] = 0;
                    for (int x = i - 1; x <= i + 1; x++)
                        for (int y = j - 1; y <= j + 1; y++)
                            if (x >= 0 && x < n && y >= 0 && y < n && str[x][y] == '*')
                                g[i][j]++;
                }

        
        int res = 0;

        // 枚举所有的未被处理的0
        for (int i = 0; i < n; i ++ )
            for (int j = 0; j < n; j ++ )
                // 如果当前位置是0
                if (!g[i][j]) {
                    res ++ ;
                    dfs(i, j);
                }

        // 统计所有剩余1-8的数量
        for (int i = 0; i < n; i ++)
            for (int j = 0; j < n; j ++)
                if(g[i][j] != -1) 
                    res ++ ;

        printf("Case #%d: %d\n", cases, res);
    }

    return 0;
}

# 并查集

AcWing 528. 奶酪

  • 数据范围：空洞的数量平方 1000^2
  • 搜索球心，无向图找连通
    • 图的遍历：DFS，BFS
    • 并查集
  • 判断相交 / 相切
#include <algorithm>
#include <iostream>
#include <cstring>

using namespace std;

const int N = 1010;
typedef long long LL;

int n, h, r;
// 空洞
struct Sphere {
    int x, y, z; // 球心坐标
}q[N];
int p[N]; // 存储并查集

// 在并查集中，每个集合都被一个树形结构表示，其中每个节点指向其父节点，直到根节点，根节点指向自己。
// 找到元素 x 所在的集合的根节点，并且进行路径压缩
int find(int x) {
    if (p[x] != x) p[x] = find(p[x]);
    return p[x];
}

int main() {
    // 测试数据组别
    int T;
    scanf("%d", &T);

    while (T -- )
    {
        scanf("%d%d%d", &n, &h, &r);

        for (int i = 0; i <= n + 1; i ++ ) p[i] = i;

        for (int i = 1; i <= n; i ++ ) 
        {
            int x, y, z;
            scanf("%d%d%d", &x, &y, &z);
            q[i] = {x, y, z};

            
            // 空洞和下边界连通
            if (abs(z) <= r) p[find(i)] = find(0);

            // 空洞和上边界连通
            if (abs(z - h) <= r) p[find(i)] = find(n + 1);
        }
        
        for (int i = 1; i <= n; i ++ )
            for (int j = i + 1; j <= n; j ++ ) 
            {
                LL dx = q[i].x - q[j].x;
                LL dy = q[i].y - q[j].y;
                LL dz = q[i].z - q[j].z;

                // 处理空洞之间的连通关系
                if (dx * dx + dy * dy + dz * dz <= 4 * (LL)r * r)
                    p[find(i)] =  find(j); // 两点连通
            }
        // 判断是否存在路径连接下边界和上边界
        if (find(0) == find(n + 1)) puts("Yes");
        else puts("No");
    }

    return 0;
}

# 哈希表

AcWing 1402. 星空之夜（每日一题）

  • 搜索星群 = 找连通块：Flood Fill / 并查集
  • 区分星群：暴力 / 哈希 优化：计算两两之间直线距离的总和
  • 字典序最小：找到连通块后，未赋值，赋给当前未用的最小字典序的字母
  • 浮点数判断不能写 ==，因为 float，double 都是近似表达
    • 通过 | A - B| < e 比较结果判断，若 A < B - e，A < B；若 A < B + e，A ≤ B；
#include <algorithm>
#include <cstring>
#include <iostream>
#include <cmath>

using namespace std;

#define x first
#define y second

typedef pair<int, int> PII;

const int N = 110, M = 170;
const double eps = 1e-8;

int n, m;
char g[N][N];
PII q[M];
int top = 0, cnt = 0;
double hash_val[30];

void dfs(int a, int b) 
{
    q[top ++ ] = {a, b};
    g[a][b] = '0';

    for (int x = a - 1; x <= a + 1; x ++ )
        for (int y = b - 1; y <= b + 1; y ++ )
            if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == '1')
                dfs(x, y);
}

double get_distance(PII& a, PII& b)
{
    double dx = a.x - b.x, dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy); // cmath库
}

double get_hash() 
{
    double sum = 0;
    for (int i = 0; i < top; i ++ )
        for (int j = 0; j < i; j ++ )
            sum += get_distance(q[i], q[j]); // PII* q[i] = &a; PII* q[j] = &b;
    return sum;
}

char get_id() 
{
    double val = get_hash();
    for (int i = 0; i < cnt; i ++ )
        if (abs(hash_val[i] - val) < eps)
            return 'a' + i;
    hash_val[cnt ++ ] = val;
    return 'a' + cnt - 1;
}

int main() {
    scanf("%d%d", &m, &n);
    for (int i = 0; i < n; i ++ ) scanf("%s", g[i]);

    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            // 当前位置有一颗星星
            if(g[i][j] == '1')
            {
                top = 0;
                dfs(i ,j);
                char id = get_id();
                for (int k = 0; k < top; k ++ )
                    g[q[k].x][q[k].y] = id;
            }

    for (int i = 0; i < n; i ++ ) puts(g[i]);

    return 0;
}
